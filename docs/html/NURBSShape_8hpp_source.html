57struct nurbs_shape_internal::NURBSShapeHelper<vec_t, Vec<typename vec_t::scalar_t, 2>>{
58 typedef Vec<typename vec_t::scalar_t, 2> param_vec_t;
59 typedef typename vec_t::scalar_t scalar_t;
60 enum { dim = vec_t::dim,
61 proj_dim = dim + 1,
62 param_dim = param_vec_t::dim};

65 static DomainDiscretization<vec_t> discretizeBoundaryWithDensity(
66 const NURBSShape<vec_t, param_vec_t>& shape,
67 const std::function<scalar_t(vec_t)>& h, int type) {
68 if (type == 0) type = -1;
69 std::mt19937 gen(shape.seed_);
70
71 GeneralSurfaceFill<vec_t, param_vec_t> gsf;
72 gsf.seed(shape.seed_).maxPoints(shape.max_points).proximityTolerance(shape.zeta)
73 .numSamples(shape.n_samples);
74
75 // Global domain and KDTree.
76 DomainDiscretization<vec_t> domain(shape);
77 KDTreeMutable<vec_t> tree_global;
78
79 // Discretize every patch.
80 for (const NURBSPatch<vec_t, param_vec_t>& patch : shape.patches) {
81 BoxShape<param_vec_t> bs = patch.getDomain();
82 DomainDiscretization<param_vec_t> local_param_d(bs);
83
84 // Discretize boundaries first.
85 Range<NURBSPatch<vec_t, Vec<scalar_t, 1>>> boundaries = patch.getBoundaries();
86 int i = -1;
87 for (NURBSPatch<vec_t, Vec<scalar_t, 1>>& boundary : boundaries) {
88 i++;
89 boundary.computeDerivativeStructure();
90
91 KDTreeMutable<vec_t> tree_local; // Local KD-Tree should be per-boundary.
92 // Generate seed node.
93 BoxShape<Vec<scalar_t, 1>> boundary_bs = boundary.getDomain();
94 Vec<scalar_t, 1> t_min = boundary_bs.beg();
95 Vec<scalar_t, 1> t_max = boundary_bs.end();
96
97 std::uniform_real_distribution<> dis(t_min(0), t_max(0));
98 Vec<scalar_t, 1> t_seed = Vec<scalar_t, 1>(dis(gen));
99
100 // Insert into local structures.
101 param_vec_t param_seed = patch.getPatchParameterFromBoundaryParameter(t_seed,
102 i, shape.epsilon);
103 scalar_t w_seed;
104 vec_t pt_seed = boundary.evaluate(t_seed, &w_seed);
105 local_param_d.addInternalNode(param_seed, 1);
106 tree_local.insert(pt_seed);
107
108 // Insert into global structures.
109 scalar_t check_radius_seed = h(pt_seed);
110 scalar_t d_sq = tree_global.size() == 0 ? 10 * check_radius_seed * check_radius_seed
111 : tree_global.query(pt_seed).second[0];
112 if (d_sq >= (shape.zeta * check_radius_seed) * (shape.zeta * check_radius_seed)) {
113 vec_t normal_seed = surface_fill_internal::
114 compute_normal(patch.jacobian(param_seed, pt_seed, w_seed));
115 domain.addBoundaryNode(pt_seed, type, normal_seed);
116 tree_global.insert(pt_seed);
117 }
118
119 // General surface fill the boundary.
120 int cur_node = local_param_d.size() - 1;
121 int end_node = local_param_d.size();
122 while (cur_node < end_node && end_node < shape.max_points) {
123 param_vec_t param = local_param_d.pos(cur_node);
124 Vec<scalar_t, 1> t = patch.getBoundaryParameterFromPatchParameter(param, i);
125 vec_t pt, der;
126 std::tie(pt, der) = boundary.evaluatePointAndJacobian(t);
127
128 std::array<Vec<scalar_t, 1>, 2> candidates{Vec<scalar_t, 1>(1),
129 Vec<scalar_t, 1>(-1)};
130
131 // Filter candidates.
132 scalar_t alpha = h(pt) / der.norm();
133 for (const auto& u_cand : candidates) {
134 Vec<scalar_t, 1> t_new = t + alpha * u_cand; // Generate candidate.
135 param_vec_t param_new = patch.getPatchParameterFromBoundaryParameter(
136 t_new, i, shape.epsilon);
137 if (!local_param_d.contains(param_new)) continue;
138
139 scalar_t w_new;
140 vec_t pt_new = boundary.evaluate(t_new, &w_new);
141
142 // Check radius must be new radius.
143 scalar_t check_radius = (pt - pt_new).norm();
144
145 // Check for local insertion.
146 scalar_t d_sq_loc = tree_local.query(pt_new).second[0];
147 if (d_sq_loc >= (shape.zeta * check_radius) *
148 (shape.zeta * check_radius)) {
149 local_param_d.addInternalNode(param_new, 1);
150 tree_local.insert(pt_new);
151 end_node++;
152
153 // Check for global insertion.
154 scalar_t d_sq_global = tree_global.query(pt_new).second[0];
155 if (d_sq_global >= (shape.zeta * check_radius) *
156 (shape.zeta * check_radius)) {
157 vec_t normal_new = surface_fill_internal::
158 compute_normal(patch.jacobian(param_new, pt_new, w_new));
159 domain.addBoundaryNode(pt_new, type, normal_new);
160 tree_global.insert(pt_new);
161 }
162 }
163 }
164 cur_node++;
165 }
166 }
167
168 // Random another parameter seed around the middle of the domain, so that fill is
169 // successful even if patches overlap.
170 std::uniform_real_distribution<> dis(0.0, 1.0);
171 param_vec_t param_another = bs.beg() + 0.25 * (bs.end() - bs.beg()) +
172 dis(gen) * 0.25 * (bs.end() - bs.beg());
173 local_param_d.addInternalNode(param_another, 1);
174
175 scalar_t w_another;
176 vec_t pt_another = patch.evaluate(param_another, &w_another);
177 scalar_t check_radius_another = h(pt_another);
178 scalar_t d_sq_another = tree_global.query(pt_another).second[0];
179 if (d_sq_another >= (shape.zeta * check_radius_another) *
180 (shape.zeta * check_radius_another)) {
181 vec_t normal_another = surface_fill_internal::
182 compute_normal(patch.jacobian(param_another, pt_another, w_another));
183 domain.addBoundaryNode(pt_another, type, normal_another);
184 tree_global.insert(pt_another);
185 }
186
187 // Surface fill.
188 auto param = [&patch](const param_vec_t& t) {
189 return patch.evaluatePointAndJacobian(t);
190 };
191 gsf.fillParametrization(domain, local_param_d, param, h, tree_global, type);
192 }
193
194 return domain;
195 }
